# npm-yarn-command

npm install yarn --global

# Install packages
npm install
yarn

# Add package
npm install react --save
npm i react -S
yarn add react

# Add dev package
npm i react -D
yarn add react -D

# Remove package
npm remove react --save
npm r react -S
yarn remove react

# Add global package
npm install nodemon --global
npm i nodemon -g
yarn global add nodemon

# run script
npm run build
yarn run build
yarn build
